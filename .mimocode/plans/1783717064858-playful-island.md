# Cascading Key Selector for Nested Objects

## Requirement

When a key contains an object or array, the user should be able to drill down:
1. Select "test" → shows internal keys "Provenance", "Quantite", "NbArticles"
2. If "Provenance" was also an object, show its internal keys
3. Continue recursively for infinite nesting

The variable_key would store the full path like "test.Provenance" or "test.Quantite".

## Solution

### Change 1: Create a `NestedKeySelector` component

```tsx
function NestedKeySelector({
  keys,
  records,
  value,
  onChange,
  disabled,
}: {
  keys: string[];
  records: Record<string, unknown>[];
  value: string;
  onChange: (path: string) => void;
  disabled: boolean;
}) {
  // Parse the current path into segments
  const segments = value ? value.split('.') : [];
  
  // Get the value at a given path
  const getValueAtPath = (path: string): unknown => {
    const parts = path.split('.');
    let current: unknown = records[0];
    for (const part of parts) {
      if (current && typeof current === 'object' && part in (current as Record<string, unknown>)) {
        current = (current as Record<string, unknown>)[part];
      } else {
        return undefined;
      }
    }
    return current;
  };
  
  // Get keys at a given path
  const getKeysAtPath = (path: string): string[] => {
    if (!path) return keys;
    const val = getValueAtPath(path);
    if (val && typeof val === 'object' && !Array.isArray(val)) {
      return Object.keys(val as Record<string, unknown>);
    }
    return [];
  };
  
  // Build the path segments for display
  const pathSegments: string[] = [];
  let currentPath = '';
  for (let i = 0; i < segments.length; i++) {
    const segment = segments[i];
    currentPath = currentPath ? `${currentPath}.${segment}` : segment;
    pathSegments.push(currentPath);
  }
  
  // Determine which keys are objects (need further drilling)
  const isObjectKey = (path: string): boolean => {
    const val = getValueAtPath(path);
    return val !== null && val !== undefined && typeof val === 'object' && !Array.isArray(val);
  };
  
  // Build the select elements
  const selects: JSX.Element[] = [];
  
  // First select: top-level keys
  selects.push(
    <select
      key="0"
      value={segments[0] || ''}
      onChange={(e) => {
        const key = e.target.value;
        if (key) {
          // Check if this key is an object
          const val = getValueAtPath(key);
          if (val && typeof val === 'object' && !Array.isArray(val)) {
            // Object - just select the key, don't set final value yet
            onChange(key);
          } else {
            // Simple value - set the final value
            onChange(key);
          }
        } else {
          onChange('');
        }
      }}
      disabled={disabled}
      className={`${selectBase} w-44`}
    >
      <option value="">— clé JSON —</option>
      {keys.map((k) => (
        <option key={k} value={k}>{k}</option>
      ))}
    </select>
  );
  
  // Additional selects for nested objects
  for (let i = 0; i < segments.length; i++) {
    const currentPath = pathSegments[i];
    if (isObjectKey(currentPath) && i < segments.length - 1) {
      const nestedKeys = getKeysAtPath(currentPath);
      selects.push(
        <select
          key={i + 1}
          value={segments[i + 1] || ''}
          onChange={(e) => {
            const key = e.target.value;
            const newPath = `${currentPath}.${key}`;
            const val = getValueAtPath(newPath);
            if (val && typeof val === 'object' && !Array.isArray(val)) {
              // Object - continue drilling
              onChange(newPath);
            } else {
              // Simple value - set the final value
              onChange(newPath);
            }
          }}
          disabled={disabled}
          className={`${selectBase} w-44`}
        >
          <option value="">— sélectionner —</option>
          {nestedKeys.map((k) => (
            <option key={k} value={k}>{k}</option>
          ))}
        </select>
      );
    }
  }
  
  // If the last selected key is an object, add one more empty select
  if (segments.length > 0) {
    const lastPath = pathSegments[segments.length - 1];
    if (isObjectKey(lastPath)) {
      const nestedKeys = getKeysAtPath(lastPath);
      selects.push(
        <select
          key={segments.length + 1}
          value=""
          onChange={(e) => {
            const key = e.target.value;
            const newPath = `${lastPath}.${key}`;
            onChange(newPath);
          }}
          disabled={disabled}
          className={`${selectBase} w-44`}
        >
          <option value="">— sélectionner —</option>
          {nestedKeys.map((k) => (
            <option key={k} value={k}>{k}</option>
          ))}
        </select>
      );
    }
  }
  
  return (
    <div className="flex items-center gap-1 flex-wrap">
      {selects}
    </div>
  );
}
```

### Change 4: Update validation to handle nested paths

```tsx
function validateDirectKeyType(records: Record<string, unknown>[], key: string): string | null {
  if (records.length === 0) return null;
  
  const values = records.map(r => getValueAtPath(r, key));
  
  for (let i = 0; i < values.length; i++) {
    const val = values[i];
    if (Array.isArray(val)) {
      return `La clé "${key}" contient un tableau (enregistrement ${i + 1}). Utilisez le type Complex.`;
    }
    if (typeof val === "object" && val !== null && Object.keys(val).length > 1) {
      return `La clé "${key}" contient un objet avec plusieurs clés (enregistrement ${i + 1}). Utilisez le type Complex.`;
    }
  }
  
  const uniqueValues = new Set(values.map(v => JSON.stringify(v)));
  if (uniqueValues.size > 1) {
    return `La clé "${key}" a ${uniqueValues.size} valeurs différentes. Utilisez le type Complex.`;
  }
  
  return null;
}
```
